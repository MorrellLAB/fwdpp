243  /*!
244  Returns a position from a region that is chosen based on region weights.
245 
246  \precondition If recrate == 0, then beg and end cannot be empty. It is up to the calling environment to make sure
247  that this cannot happen. This is checked in debug mode via the assert macro.
248  */
249  template<typename gamete_t,
250  typename mcont_t>
251  inline result_type operator()(const gsl_rng * r,
252  const double recrate,
253  const gamete_t &,
254  const gamete_t &,
255  const mcont_t &) const
256  {
257  assert( !(recrate==0. && (beg.empty()||end.empty())) );
258  auto nbreaks = gsl_ran_poisson(r,recrate);
259  if(!nbreaks) return {};
260 
261  result_type rv;
262  for( unsigned i=0;i<nbreaks;++i )
263  {
264  size_t region = gsl_ran_discrete(r,lookup.get());
265  rv.push_back(gsl_ran_flat(r,beg[region],end[region]));
266  }
267  std::sort(rv.begin(),rv.end());
268  rv.push_back(std::numeric_limits<double>::max());
269  return rv;
270  }
