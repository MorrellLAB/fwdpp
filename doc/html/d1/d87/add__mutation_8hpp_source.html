28  /*!
29  \brief Helper function for implementing KTfwd::add_mutation
30 
31  If \a new_mutation already exists in \a mutations, its location is returned.
32 
33  Otherwise, this function puts the new mutation into the mutations container and
34  updates other objects as needed.
35 
36  \return A key telling where \a new_mutation exists within \a mutations.
37 
38  \note This function requires that operator== be defined for the mutation type.
39 
40  \note In the event that \a new_mutation is added to \a mutations, the count for that
41  mutations will be zero.
42 
43  \ingroup sugar
44  */
45  {
46  //If the mutation already exists, let's not create it...
47  auto mitr = std::find(std::begin(mutations),std::end(mutations),new_mutation);
48  if(mitr!=mutations.end()) return std::size_t(std::distance(std::begin(mutations),mitr));
49 
50  /*
51  If we are here, then new_mutation does not currently exist, so we have to add it.
52 
53  We do this using fwdpp's recycling method. But, we don't need to create recycling bin,
54  as we're just looking to add one mutation. Instead, we can simply search for an extinct
55  mutation.
56  */
57  //find an extinct mutation, if one exists
58  auto extinct_mut = std::find(std::begin(mcounts),std::end(mcounts),0);
59  std::size_t mindex=mcounts.size(); //this will be the correct value if we use the else block below
60  if( extinct_mut != std::end(mcounts) ) //then we can recycle
61  {
62  auto dist=std::distance(std::begin(mcounts),extinct_mut);
63  mutations[dist] = std::move(new_mutation); //move new mutation into place
64  mindex=std::size_t(dist); //update our value
65  }
66  else
67  {
68  //cannot recycle, so add it to end
69  mutations.emplace_back(std::move(new_mutation));
70  mcounts.push_back(0); //Add a place for this variant
71  }
72  return mindex;
