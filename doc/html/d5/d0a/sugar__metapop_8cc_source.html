126 {
127  //Evolve for 10 generations
128  std::vector<std::function<double (const poptype::gamete_t &,
129  const poptype::gamete_t &,

136  std::vector<double> wbar = KTfwd::sample_diploid(rng.get(),
137  pop.gametes,

139  pop.mutations,
140  pop.mcounts,
141  &pop.Ns[0],
142  0.005,
143  std::bind(KTfwd::infsites(),std::placeholders::_1,std::placeholders::_2,rng.get(),std::ref(pop.mut_lookup),generation,
144  0.005,0.,[&rng](){return gsl_rng_uniform(rng.get());},[](){return 0.;},[](){return 0.;}),
145  std::bind(KTfwd::poisson_xover(),rng.get(),0.005,0.,1.,
146  std::placeholders::_1,std::placeholders::_2,std::placeholders::_3),
147  fitness_funcs,
148  std::bind(migpop,std::placeholders::_1,rng.get(),0.001),
149  pop.neutral,pop.selected);
150  KTfwd::update_mutations(pop.mutations,pop.fixations,pop.fixation_times,pop.mut_lookup,pop.mcounts,generation,4000);
151  }
