22 struct additive_over_loci
23 {
24  template<typename gamete_t,
25  typename mcont_t>
26  inline double operator()(const gamete_t & g1,
27  const gamete_t & g2,
28  const mcont_t & mutations,
29  const unsigned K) const noexcept
30  /*
31  The fitness model is additive over loci, multiplicative within loci
32  */
33  {
34  double rv = 0.0;
35  for(unsigned i=0;i<K;++i)
36  {
37  /*
38  Find the iterators corrsponding to locus 1 in g1 and g2.
39  Because gamete store mutation keys sorted by mutation position,
40  we can use efficient binary searches.
41 
42  We want the positions of all mutations such that i <= p < i+1,
43  which correspond to lower_bound and upper_bound, respectively.
44  */
45  auto stop1 = std::upper_bound(g1.smutations.begin(),g1.smutations.end(),double(i+1),
46  [&mutations](const double val, const std::size_t i) {
47  return val < mutations[i].pos;
48  });
49  auto start1 = std::lower_bound(g1.smutations.begin(),
50  stop1,double(i),
51  [&mutations](const std::size_t i,const double val) {
52  return mutations[i].pos < val;
53  });
54  auto stop2 = std::upper_bound(g2.smutations.begin(),g2.smutations.end(),double(i+1),
55  [&mutations](const double val, const std::size_t i) {
56  return val < mutations[i].pos;
57  });
58  auto start2 = std::lower_bound(g2.smutations.begin(),
59  stop2,double(i),
60  [&mutations](const std::size_t i,const double val) {
61  return mutations[i].pos < val;
62  });
63 
64  rv += KTfwd::multiplicative_diploid()(start1,stop1,start2,stop2,mutations,2.0) - 1.0;
65  }
66  return std::max(1.0 + rv,0.0);
67  }
