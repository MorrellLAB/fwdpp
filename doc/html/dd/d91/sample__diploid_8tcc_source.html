115  /*
116  The mutation and gamete containers contain both extinct and extant objects.
117  The former are useful b/c the represent already-allocated memory. The library
118  uses these extinct objects to 'recycle' them into new objects. The function calls
119  below create FIFO queues of where extinct objects are. These queues are passed to
120  mutation and recombination functions and used to decide if recyling is possible or
121  if a new object needs to be 'emplace-back'-ed into a container.
122 
123  The type of the FIFO queue is abstracted with the name KTfwd::fwdpp_internal::recycling_bin_t,
124  which is a C++11 template alias.
125 
126  The details of recycling are implemented in fwdpp/internal/recycling.hpp
127  */
128  auto mut_recycling_bin = fwdpp_internal::make_mut_queue(mcounts);
129  auto gam_recycling_bin = fwdpp_internal::make_gamete_queue(gametes);

133  //create a vector to store fitnesses:
134  std::vector<double> fitnesses(diploids.size());
135  double wbar = 0.; //pop'n mean fitness
136  for( uint_t i = 0 ; i < N_curr ; ++i )
137  {
138  /*
139  Set the count of each gamete to 0.
140 
141  Yes, in the case of > 1 diploid having the exact same gamete index,
142  this is done multiple times.
143  */
144  gametes[diploids[i].first].n=gametes[diploids[i].second].n=0;
145  /*
146  Assign fitness to the i-th individual.
147 
148  ff is a "fitness function", which returns a double. For examples, see
149  KTfwd::multiplicative_diploid, which is a "standard" type of fitness function
150  used in population genetics. "Standard" types of models are defined in
151  fwdpp/fitness_models.hpp.
152 
153  The library supports various types of function signatures for fitness models.
154  In order to get the final type of call right, we use a dispatch method, which is
155  the call below. Note that this dispatch happens at compile-time rather than at run-time.
156  Essentially, the compiler has to work out what the final call will be, and this dispatch operation
157  itself is inlined out. The reason why this is compile-time is b/c the dispatch depends on
158  whether or not the diploid is a "custom type", which is a property of a simulation and it is
159  not possible to mix custom and non-custom diploids in the same simulation.
160 
161  The dispatch itself is implemented in fwdpp/internal/diploid_fitness_dispatch.hpp
162 
163  See the tutorial on custom diploids for more details.
164  */
165  fitnesses[i] = fwdpp_internal::diploid_fitness_dispatch(ff,diploids[i],gametes,mutations,
166  typename traits::is_custom_diploid_t<diploid_geno_t>::type());
167  wbar += fitnesses[i];
168  }
169  wbar /= double(diploids.size());
170 #ifndef NDEBUG

175  This is a lookup table for rapid sampling of diploids proportional to their fitnesses.
176  This is a unique_ptr wrapper around an object from the GNU Scientific Library. A custom deleter
177  is required to make this work, which is why there is no cleanup call down below.
178  */
179  fwdpp_internal::gsl_ran_discrete_t_ptr lookup(gsl_ran_discrete_preproc(N_curr,fitnesses.data()));
180  const auto parents(diploids); //Copy the parents, which is trivally fast for the vast majority of use cases.
181 
182  //Change the population size
183  if( diploids.size() != N_next)
184  {
185  diploids.resize(N_next);
186  }
187  assert(diploids.size()==N_next);

189  //Fill in the next generation!
190  for(auto & dip : diploids)
191  {
192  //Choose parent 1 based on fitness
193  auto p1 = gsl_ran_discrete(r,lookup.get());
194  //If inbred (w/probability f2), parent2 = parent1, else choose again based on fitness
195  auto p2 = (f==1. || (f>0. && gsl_rng_uniform(r) < f)) ? p1 : gsl_ran_discrete(r,lookup.get());
196  assert(p1<parents.size());
197  assert(p2<parents.size());
198 
199  /*
200  These are the gametes from each parent.
201  This is a trivial assignment if keys.
202  */
203  auto p1g1 = parents[p1].first;
204  auto p1g2 = parents[p1].second;
205  auto p2g1 = parents[p2].first;
206  auto p2g2 = parents[p2].second;
207 
208  /*
209  The offspring will inherit some manipulation of p1g1 and p1g2.
210  The next two lines do "Mendel".
211  */

213  if(gsl_rng_uniform(r)<0.5) std::swap(p2g1,p2g2);

215  /*
216  Recombination: the offspring's first gamete will be a recombinant of
217  p1g1 and p1g2. Likewise for the second gamete.
218 
219  Internally, the recombination function does the following:
220  1. Use rec_pol, a programmer-defined policy, to determine the positions of breakpoints.
221  2. Create new mixes of the parental gametes into the pre-allocated containers 'neutral' and 'selected'
222 
223  The implementation is in fwdpp/recombination.[hpp|tcc] and the gory details
224  are in fwdpp/internal/recombination_common.hpp and fwdpp/internal/rec_gamete_updater.hpp.
225 
226  Briefly, recombination is implemented via a series of sequential updates to iterators using binary
227  searches (std::upper_bound).
228  */
229  dip.first = recombination(gametes,gam_recycling_bin,
230  neutral,selected,rec_pol,p1g1,p1g2,mutations).first;
231  dip.second = recombination(gametes,gam_recycling_bin,
232  neutral,selected,rec_pol,p2g1,p2g2,mutations).first;
233 
234  //update gamete counts
235  gametes[dip.first].n++;
236  gametes[dip.second].n++;
237 
238  /*
239  Now, add new mutations to the first and second gamete of the offspring.
240 
241  In fwdpp, mutation keys in gametes are "less-than" sorted according to mutation position.
242 
243  The keys to mutations are inserted into the correct place which is found by a binary search (std::upper_bound).
244 
245  I think that there are ways to improve the speed of this part of the code in order to reduce calls
246  to memmove/memcpy, but profiling suggests that the payoff would be small. (Note that memmove/memcyp is
247  never called direclty. Rather, they are called indirectly via the insert() member function of the relevant
248  container type.)
249 
250  The implementation details are found in fwdpp/mutation.hpp
251  */
252  dip.first = mutate_gamete_recycle(mut_recycling_bin,gam_recycling_bin,r,mu,gametes,mutations,dip.first,mmodel,gpolicy_mut);
253  dip.second = mutate_gamete_recycle(mut_recycling_bin,gam_recycling_bin,r,mu,gametes,mutations,dip.second,mmodel,gpolicy_mut);
254 
255  assert( gametes[dip.first].n );
256  assert( gametes[dip.second].n );
257  }
258  assert(check_sum(gametes,2*N_next));
259 #ifndef NDEBUG
260  for(const auto & dip : diploids)
261  {
262  assert(gametes[dip.first].n>0);
263  assert(gametes[dip.first].n<=2*N_next);
264  assert(gametes[dip.second].n>0);
265  assert(gametes[dip.second].n<=2*N_next);
266  }
267 #endif
268  /*
269  At the end of the above loop, we have a bunch of new diploids
270  that are all recombined and mutated sampling of the parental generation.
271 
272  Our problem is that we no longer know how many times each mutation is present, which
273  is corrected by the following call.
274 
275  Although the implementation of process_gametes is super-trivial, it is actually the
276  most computationally-expensive part of a simulation once mutation rates are large.
277 
278  Further, the function is hard to optimize. Recall that gametes store mutations in order
279  according to position. Thus, when we go from a gamete to a position in mcounts, we are
280  accessing the latter container out of order with respect to location in memory. process_gametes
281  is thus the "scatter" part of a "scatter-gather" idiom. Modern x86 CPU have little available
282  for vectorizing such cases. I've experimented with CPU intrinsics to attempt memory prefetches,
283  but never saw any performance improvement, and the code got complex, and possibly less portable.
284 
285  The implementation is in fwdpp/internal/sample_diploid_helpers.hpp
286  */
287  fwdpp_internal::process_gametes(gametes,mutations,mcounts);
288  assert(mcounts.size()==mutations.size());
289 #ifndef NDEBUG
290  for(const auto & mc : mcounts)
291  {
292  assert(mc <= 2*N_next);
293  }
294 #endif
295  assert(popdata_sane(diploids,gametes,mutations,mcounts));
296 
297  /*
298  The last thing to do is handle fixations. In many contexts, we neither want nor need
299  to keep indexes to fixed variants in our gametes. Such decisions are implemented via
300  simple policies, which are in the variable 'mp'.
301 
302  The implementation is in fwdpp/internal/gamete_cleaner.hpp.
303 
304  The implementation is the "erase/remove idiom" (Effective STL, Item 32), but with a twist
305  that the function will exit early if there are no fixations present in the population at
306  the moment.
307 
308  Example policies are KTfwd::remove_nothing and KTfwd::remove_neutral, both found
309  in fwdpp/fwd_functional.hpp. If mp is std::true_type, then all fixations (e.g., neutral
310  and selected) will be removed from all gametes.
311  */
312  fwdpp_internal::gamete_cleaner(gametes,mutations,mcounts,2*N_next,mp);
313  return wbar;
314  }

376  sample_diploid(const gsl_rng * r,

378  metapop_diploid_vector_type < diploid_vector_type<diploid_geno_t,diploid_vector_type_allocator>,metapop_diploid_vector_type_allocator > & diploids,
379  mutation_cont_type<mutation_type,mutation_cont_type_allocator > & mutations,
380  std::vector<uint_t> & mcounts,

382  const uint_t * N_next,
383  const double & mu,
384  const mutation_model & mmodel,
385  const recombination_policy & rec_pol,
386  const diploid_fitness_function_container & ffs,
387  const migration_policy & mig,
388  typename gamete_type::mutation_container & neutral,
389  typename gamete_type::mutation_container & selected,

391  const mutation_removal_policy & mp,
392  const gamete_insertion_policy & gpolicy_mut)
393  {
394  //get the fitnesses for each diploid in each deme and make the lookup table of parental fitnesses
395  using lookup_t = fwdpp_internal::gsl_ran_discrete_t_ptr;
396  std::vector<lookup_t> lookups;
397  std::vector<double> wbars(diploids.size(),0);
398  auto mut_recycling_bin = fwdpp_internal::make_mut_queue(mcounts);
399  auto gamete_recycling_bin = fwdpp_internal::make_gamete_queue(gametes);
400 
401  //get max N
402  auto mN = (!diploids.empty()) ? *std::max_element(N_curr,N_curr+diploids.size()): 0;
403 
404  std::vector<double> fitnesses(mN);
405  std::size_t popi=0;
406  for(const auto & dipvec : diploids ) //go over each container of diploids...
407  {
408  std::vector<double>::size_type i=0;
409  for(const auto & dip : dipvec) //...and each diploid
410  {
411  fitnesses[i]=fwdpp_internal::diploid_fitness_dispatch(ffs[popi],dip,gametes,mutations,typename traits::is_custom_diploid_t<diploid_geno_t>::type());
412  wbars[popi]+=fitnesses[i];
413  gametes[dip.first].n=gametes[dip.second].n=0;
414  ++i;
415  }
416  wbars[popi] /= double(dipvec.size());
417  lookups.emplace_back(lookup_t(gsl_ran_discrete_preproc(dipvec.size(),fitnesses.data())));
418  ++popi;
419  }
420 
421  assert(lookups.size() == diploids.size());
422  //copy diploids into temporary parents
423  const auto parents(diploids);
424 
425  //Update the diploids, one deme at a time
426  for(popi = 0 ; popi < diploids.size() ; ++popi)
427  {
428  auto demesize = *(N_next+popi);
429  if(demesize != *(N_curr+popi))
430  {
431  diploids[popi].resize(demesize);
432  }
433  for(auto & dip : diploids[popi])
434  {
435  /* Figure out if parent 1 is migrant or not.
436 
437  A migration policy takes the current deme (popindex) as
438  an argument. It returns popindex if there is no migration,
439  else it returns the index of the deme of a migrant parent
440  */
441  auto deme_p1 = mig(popi);
442  decltype(deme_p1) deme_p2=popi;
443 
444  //Figure out who the parents are
445  auto p1 = gsl_ran_discrete(r,lookups[deme_p1].get()),p2=p1;
446 
447  /*
448  If the individual is not inbred, then we pick a
449  deme from the migration policy for parent 2
450  */
451  if( f != nullptr && ( *(f + popi)==1. || (*(f + popi)>0. && gsl_rng_uniform(r) < *(f + popi)) ) ) //individual is inbred
452  {
453  p2=p1;
454  }
455  else
456  {
457  //apply migration policy to figure out parental deme for parent #2
458  deme_p2 = mig(popi);
459  p2 = gsl_ran_discrete(r,lookups[deme_p2].get());
460  }
461 
462  auto p1g1 = parents[deme_p1][p1].first;
463  auto p1g2 = parents[deme_p1][p1].second;
464  auto p2g1 = parents[deme_p2][p2].first;
465  auto p2g2 = parents[deme_p2][p2].second;
466 
467  if(gsl_rng_uniform(r)<0.5)std::swap(p1g1,p1g2);
468  if(gsl_rng_uniform(r)<0.5)std::swap(p2g1,p2g2);
469 
470  dip.first = recombination(gametes,gamete_recycling_bin,
471  neutral,selected,rec_pol,p1g1,p1g2,mutations).first;
472  dip.second = recombination(gametes,gamete_recycling_bin,
473  neutral,selected,rec_pol,p2g1,p2g2,mutations).first;
474 
475  gametes[dip.first].n++;
476  gametes[dip.second].n++;
477 
478  //now, add new mutations
479  dip.first = mutate_gamete_recycle(mut_recycling_bin,gamete_recycling_bin,r,mu,gametes,mutations,dip.first,mmodel,gpolicy_mut);
480  dip.second = mutate_gamete_recycle(mut_recycling_bin,gamete_recycling_bin,r,mu,gametes,mutations,dip.second,mmodel,gpolicy_mut);
481  }
482  }
483  fwdpp_internal::process_gametes(gametes,mutations,mcounts);
484  assert(mcounts.size()==mutations.size());
485  fwdpp_internal::gamete_cleaner(gametes,mutations,mcounts,
486  2*std::accumulate(N_next,N_next+diploids.size(),uint_t(0)),mp);
487  assert( check_sum(gametes,2*std::accumulate(N_next,N_next+diploids.size(),uint_t(0))) );
488  return wbars;
489  }
