10 /*!
11  \file gamete_cleaner.hpp
12 
13  This file handles the "pruning" of fixations from gametes.
14 
15  Whether or not any pruning happens depends on policies.
16 
17  Version 0.4.7 of the library made a major change resulting in improved
18  run-time performance, especially for low mutation rates.
19 
20  Consider the case of neutral mutations occurring at rate mu (per site,
21  per generation). The rate of fixation is also mu. Thus, when mu is low,
22  the rate of fixation is low, and the expected time in generations between
23  fixations is 1/mu. In other words, calling gamete_cleaner every generation
24  is an unnecessary run-time cost. Now, we check if any fixations exist, and return
25  if there are none. We check separately for neutral- and non-neutral fixations.
26 
27  Naively, we could simply scan the mutations container of the entire simulation for fixations.
28  But, we can do better than that because:
29 
30  1. Fixations are, by definition, found in all gametes. Thus, it suffices to check the
31  first (extant) gamete.
32 
33  2. When we find the first fixation in a gamete, it is by definition the fixation with the smallest
34  position, out of all fixations. We can store its value and use it to look up the first fixation
35  in all of the remaining gametes. Searching for the first fixation in this way avoids cache misses that
36  are unavoidable when we do out-of-order lookups in "mcounts" for the remaining fixations.
37 */

149  The two overloads of gamete_cleaner dispatch the above policies into this function.
150  */
151  {
152  auto extant_gamete = next_extant_gamete(gametes.begin(),gametes.end());
153  if(extant_gamete==gametes.end()) return;
154  const auto fixation_n = ff(extant_gamete->mutations);
155  bool neutral_fixations_exist = (fixation_n!=extant_gamete->mutations.cend());
156  const auto fixation_s = ff(extant_gamete->smutations);
157  bool selected_fixations_exist = (fixation_s!=extant_gamete->smutations.cend());
158  if(!neutral_fixations_exist && !selected_fixations_exist) return;
159 
160  const auto gend = gametes.end();
161  //Assign values to avoid tons of de-referencing later
162  const auto fixation_n_value = (fixation_n==extant_gamete->mutations.cend()) ? typename decltype(fixation_n)::value_type() : *fixation_n;
163  const auto fixation_s_value = (fixation_s==extant_gamete->smutations.cend()) ? typename decltype(fixation_s)::value_type() : *fixation_s;
164  while(extant_gamete < gend)
165  {
166  if(neutral_fixations_exist)
167  {
168  iw(extant_gamete->mutations,fixation_n_value);
169  }
170  if(selected_fixations_exist)

172  iw(extant_gamete->smutations,fixation_s_value);
173  }
